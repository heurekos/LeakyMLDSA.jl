Real World

Evaluation of our attack on real ML-DSA signatures generated by OpenSSL.

This section assumes a Linux system with all prerequisites for building OpenSSL in place.

Helper Functions

We begin by defining a function load_signature that, for indices $i \in \{1,\dotsc,\ell\}$ and $j \in \{1,\dotsc,256\}$, extracts from an ML-DSA signature a simplified signature $(c, z) \in \{-1,0,1\}^{256} \times (-\gamma+\beta,\gamma-\beta)$. For further details, see Section 3.2 of IACR ePrint 2026/580 and the ML-DSA specification.

using LeakyMLDSA
using CryptographicHashFunctions: digest!, shake256

struct RealMLDSA
    n::Int; log_gamma::Int; ell::Int; eta::Int8; tau::Int; strength::Int; sig_length::Int
end

const RealMLDSA44 = RealMLDSA(256, 17, 4, 2, 39, 32, 2420)
const RealMLDSA65 = RealMLDSA(256, 19, 5, 4, 49, 48, 3309)
const RealMLDSA87 = RealMLDSA(256, 19, 7, 2, 60, 60, 4627)

function load_signature(params::RealMLDSA, bytes::AbstractVector{UInt8}, i::Int, j::Int)
    (; n, log_gamma, strength) = params

    c = sparsify(negacyclic_shift(sample_in_ball(params, @view(bytes[1:strength])), j))
    z = unpack_z(params, @view(bytes[(strength + 1):end]), i, j)

    return (c, z)
end

function sparsify(c::AbstractVector)
    n = length(c)
    return LeakyMLDSA.SparseSignVector(
        [idx for idx in 0:(n - 1) if c[idx + 1] > 0],
        [idx for idx in 0:(n - 1) if c[idx + 1] < 0],
    )
end

function negacyclic_shift(c::Vector, j::Int)
    n = length(c)
    return [[c[j - j_ + 1] for j_ in 1:j]; [-c[n + j - j_ + 1] for j_ in (j + 1):n]]
end

function sample_in_ball(params::RealMLDSA, seed::AbstractVector{UInt8})
    (; n, tau) = params
    xof = shake256(seed)
    bits = BitVector(((b >> i) & 1) == 1 for b in digest!(xof, 8) for i in 0:7)
    c = zeros(Int, n)
    for i in (n - tau):(n - 1)
        local j; while (j = digest!(xof, 1)[]) > i end
        (c[i + 1], c[j + 1]) = (c[j + 1], bits[i + tau - n + 1] ? -1 : 1)
    end
    return c
end

function unpack_z(params::RealMLDSA, bytes::AbstractVector{UInt8}, i::Int, j::Int)
    (; n, log_gamma) = params
    (idx_i, idx_j) = ((i - 1) * cld(n * (log_gamma + 1), 8), (j - 1) * (log_gamma + 1))
    bytes = @view bytes[(idx_i + (idx_j >> 3) + 1):end]
    window = Int(bytes[1]) | (Int(bytes[2]) << 8) | (Int(bytes[3]) << 16)
    return (1 << log_gamma) - ((window >> (idx_j & 7)) & ((1 << (log_gamma + 1)) - 1))
end

Store Signatures

Next, we define a function that builds a patched version of OpenSSL and generates real signatures:

function generate_real_signatures(algorithm::AbstractString, count::Int;
    ossldir = mktempdir(),
    osslcommit = "openssl-3.6.2",
    osslpatch = """

        +    printf("\\n");
        +}
        +
         /*
          * @brief Initialize a Signature object by pointing all of its objects to
          * preallocated blocks. The values passed for hint, z and
        @@ -290,6 +303,7 @@ static int ml_dsa_sign_internal(const ML_DSA_KEY *priv,

                         | constant_time_lt(params->omega, h_ones)))
                     continue;
        +        print_polyvector(&y);
                 ret = ossl_ml_dsa_sig_encode(&sig, params, out_sig);
                 break;
             }
        """
)
    params = Dict(
        "ML-DSA-44" => RealMLDSA44,
        "ML-DSA-65" => RealMLDSA65,
        "ML-DSA-87" => RealMLDSA87,
    )[algorithm]

    (keyfile, msgfile, sigfile, sigsfile, leaksfile) =
        (joinpath(ossldir, "$algorithm.$ext") for ext in ("key", "msg", "sig", "sigs", "leaks"))

    osslcmd = setenv(Cmd([joinpath(ossldir, "apps/openssl")]), "LD_LIBRARY_PATH" => ossldir)

    if !isfile(osslcmd.exec)
        mkpath(ossldir)
        @info "cloning and building patched OpenSSL"
        cd(ossldir) do; redirect_stdout(devnull) do; redirect_stderr(devnull) do
            run(`git clone https://github.com/openssl/openssl.git .`)
            run(`git checkout $osslcommit`)
            run(pipeline(`git apply -`; stdin = IOBuffer(osslpatch)))
            run(`./Configure`)
            run(`make -j$(Sys.CPU_THREADS)`)
        end; end; end
        @assert isfile(osslcmd.exec)
    end

    @info "generating key"
    run(`$osslcmd genpkey -algorithm $algorithm -out $keyfile`)

    @info "generating signatures"
    open(sigsfile, "w") do sigsio; open(leaksfile, "w") do leaksio; redirect_stdout(leaksio) do
        for _ in 1:count
            write(msgfile, rand(UInt8, 32))
            run(`$osslcmd pkeyutl -sign -inkey $keyfile -in $msgfile -out $sigfile`)
            write(sigsio, read(sigfile))
        end
    end; end; end

    return (; params, keyfile, sigsfile, leaksfile)
end

The returned file sigsfile points to a binary file containing a sequence of ML-DSA signatures, and leaksfile points to a text file in which each line lists the coefficients of the mask vector $\boldsymbol{y}$, witch each coefficient in $[-\gamma,\gamma)$ and separated by spaces.

Load Signatures

Finally, we define a function that parses sigsfile and leaksfile. To model single-bit leakage, we extract only the bit of $\boldsymbol{y}$ specified by the arguments i, j, and lambda.

function load_leaky_signatures(;
    params::RealMLDSA, sigsfile::AbstractString, leaksfile::AbstractString,
    i::Int, j::Int, lambda::Int, filter_informative::Bool = true,
)
    (; n, log_gamma, eta, tau, sig_length) = params

    mldsa = MLDSA(; gamma = 1 << log_gamma, eta, tau, lambda)

    (cs, zs, leaks) = (LeakyMLDSA.SparseSignVector[], Int[], BitVector())
    open(sigsfile) do sigsio; open(leaksfile) do leaksio
        n_sig = 1
        while !eof(sigsio) && !eof(leaksio)
            (c, z) = load_signature(params, read(sigsio, sig_length), i, j)
            if !filter_informative || is_informative(mldsa, z)
                push!(cs, c)
                push!(zs, z)

                # Extract the "leaked" bit; adjust the next two lines for other formats,
                # e.g., when leakfile contains only single leaked bits.
                y = parse(Int, split(readline(leaksio))[(i - 1) * n + j])
                push!(leaks, (y >> lambda) & 1)

                n_sig += 1
            end
        end
    end; end
    sigs = Signatures(mldsa, cs, zs, leaks)

    return sigs
end

Key Recovery

Putting everything together, we now recover the $i$-th component of a real ML-DSA signing key $\boldsymbol{s}_1$:

(; params, sigsfile, leaksfile) = generate_real_signatures("ML-DSA-44", 1500)
sigs = load_leaky_signatures(; params, sigsfile, leaksfile, i = 2, j = 145, lambda = 4)

@info "running attack"
result = run_attack(NLPAttack(sigs))

println("violated signatures: ", sig_violations(result))
println("recovered component of s_1: [", join(key_guess(result), ", "), "]")
